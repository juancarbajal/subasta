/*-----------------------------------------------------------------------------------------------
 * Galeria 3D para la pagina de inicio
 * @submodule gallery3D
 * @main default
 *//*-------------------------------------------------------------------------------------------*/
yOSON.AppCore.addModule('gallery3D', function(Sb){
    var dom = {
        carousel:$("#carousel"),
        btnLeft:$("#carousel-btn-left"),
        btnRight: $("#carousel-btn-right"),
        btnMedia:$("#btn-media")
    },
    gallery3D = function(){
        var optionsCarousel = {trackerSummation:false,stopOnHover:true};
        dom.carousel.featureCarousel(optionsCarousel);

        dom.btnLeft.bind('click',function(){
            dom.carousel.next();    
        });
        dom.btnRight.bind('click',function(){
            dom.carousel.prev();    
        });
        dom.btnMedia.live('click',function(){
            if($(this).hasClass('btn-media-pause')){
                $(this).animate({backgroundPosition:'-1px -20px'});
                $(this).removeClass('btn-media-pause');
                $(this).addClass('btn-media-play');
                dom.carousel.pause();
            }else if($(this).hasClass('btn-media-play')){
                $(this).animate({backgroundPosition:'-1px -1px'});
                $(this).removeClass('btn-media-play');
                $(this).addClass('btn-media-pause');
                dom.carousel.start();
            }  
        });

    };
    return {
    /*Public Constructor : Inicializa el modulo (Es el que se ejecuta en el schema @modules)*/               
        init: function(oParams){
            gallery3D();
        },
        /*Constructor : Inicializa el modulo (Es el que se ejecuta en el schema @modules).*/
        destroy: function(){ /*Como destruir instacia de este modulo aqui*/ }
    };
}, ['libs/plugins/jquery.featureCarousel.js', 'plugins/feature-carousel.css']);



/**
* Mostrar mas categorias
* @submodule showMoreCategories
* @main default
*/
yOSON.AppCore.addModule('showMoreCategories', function(Sb){
    var dom = {contentMenu:$('.menu ul'),btnShowMore:$('.menu .show-all')},
    showMoreCategories = function(){
        var allTotal=0,firstItems=0,hidden=0;

        dom.contentMenu.each(function(index, value) {  
          var firstItems = 0,allTotal=0,hiddenItems=0;
          allTotal=$(this).height();          
          $('li:lt(5)',value).each(function(index, value) {
            firstItems=firstItems+parseInt($(value).height());
           });
          $(this).css("height",firstItems);
           hiddenItems=allTotal-firstItems;
           $(this).data('hiddenItems',hiddenItems);           
        });
        
        dom.btnShowMore.bind("click",function(e){
          e.preventDefault();
          var $that=$(this);
          utmp=$('ul',$(this).parent());
          hiddenItems=utmp.data('hiddenItems');
          if(!utmp.hasClass("view-all")){
            utmp.addClass("view-all");
            utmp.animate({
              height: '+='+hiddenItems
            },function(){
              $('html,body').animate({
                scrollTop : utmp.offset().top-80
              });
              $that.html('&rsaquo;&rsaquo; Menos Categorias');
            })
          }else{
            utmp.removeClass("view-all");
            utmp.animate({
              height: '-='+hiddenItems
            },function(){
              $that.html('&rsaquo;&rsaquo; MÃ¡s Categorias');
              $('html,body').animate({
                scrollTop: utmp.offset().top-80
              });
            })
          }
        });
    };
    return {
        init: function(){
            showMoreCategories();
        },
        destroy: function(){ /*destruir la instancia de este modulo aqui*/ }
    };
});

/**
* Enviar lista de aviso por email
* @submodule send-list-ad-email
* @main default
*/
yOSON.AppCore.addModule('send-list-ad-email', function(Sb){ 
    var dom = {sendListAdEmailLink:$('.send-list-ad-email-link'),btnClose:$('.btn-close')};
    return {
        init: function(){
          dom.sendListAdEmailLink.bind('click',function(){

               if($('.description-product input').length == 0){
                  var $avisos = $('.product-square input:checked');
               }else{
                     $avisos = $('.description-product input:checked');
               }

                if( $avisos.length > 0 ){
                    var $extra = $('.send-friend').find('.extra');
                    var $titulos = $avisos.parent().find('h3');
                    var $lista = $('.send-friend').find('.avisos-seleccionados');
                    var listaAvisos = '';
                    $titulos.each(function(){
                      var $this = $(this);
                      listaAvisos += '<li>'+ $this.text() +'</li>';
                    });
                    $lista.html( listaAvisos );
                    $extra.html('');
                    $extra.append( $avisos.clone() );
                    $.fancybox({
                        'transitionIn'   : 'none',
                        'transitionOut' : 'none',
                        'titleShow'    : false,
                        'type' : 'inline',
                        'content' :   '#enviar-mail',
                        'onComplete' : function(){
                            yOSON.AppCore.runModule('validate', {form:'#send-friend-form'});
                        }
                    });
                }else{
                    if( $('.description-product input:checked').length <= 0 ){
                      $.fancybox('<div class="message-modal"><div class="title-gen"><h3 class="title-step"><span class="icon-right"></span><p>Mensaje</p></h3></div><p>Debes seleccionar al menos un aviso de los resultados.</p></div>');
                      return false;
                    }
                }
          });

          dom.btnClose.bind('click',function(){
              $.fancybox.close();
          });

        },
        destroy: function(){ /*destruir la instancia de este modulo aqui*/ }
    };
},['libs/plugins/jquery.fancybox.js','plugins/fancybox/jquery.fancybox.css']);

/**
* Envia a la cabecera cuando se tiene una lista muy larga 
* @submodule scrollTop
* @main default
*/
yOSON.AppCore.addModule('scroll-top', function(Sb){ 
    var dom = {scrollTopLink :$('.scroll-top-link'),htmlDom:$('html'),windowDom:$(window)};
    return {
        init: function(){
            /* variables  */
            var middleDocument = dom.htmlDom.innerHeight()/2,
                positionScroll;

            /* block ocultar mostrar */
            if(middleDocument < positionScroll ){
                    dom.scrollTopLink.css('display','block');
                }

              if(middleDocument > positionScroll){
                     dom.scrollTopLink.css('display','none');
              }
            /* -- block ocultar mostrar */

            dom.windowDom.scroll(function(){
              positionScroll = dom.htmlDom.scrollTop();
              /* block ocultar mostrar */
              if(middleDocument < positionScroll ){
                    dom.scrollTopLink.css('display','block');
                }

              if(middleDocument > positionScroll){
                  dom.scrollTopLink.css('display','none');
              }
              /* -- block ocultar mostrar */

            });

            

            dom.scrollTopLink.bind('hover',function(){
              $(this).animate({opacity:0.8});
            });

          /* evento click para subir */
          dom.scrollTopLink.bind('click',function(){
            dom.htmlDom.animate({scrollTop:0},'slow');
          });
            
        },
        destroy: function(){ /*destruir la instancia de este modulo aqui*/ }
    };
});
